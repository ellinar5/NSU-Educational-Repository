SELECT PRODUCT.AMOUNT, MATERIAL.AMOUNT
FROM PRODUCT
LEFT JOIN MATERIAL ON PRODUCT.RECIPE_ID = MATERIAL.RECIPE_ID
ORDER BY
    MATERIAL.AMOUNT DESC NULLS LAST;

SELECT *
FROM PRODUCT
GROUP BY AMOUNT
ORDER BY COUNT(*)
LIMIT 1;

LEFT JOIN CATEGORY ON PRODUCT.ware = CATEGORY.ware;
